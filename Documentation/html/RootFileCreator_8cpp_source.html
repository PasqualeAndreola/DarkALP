27int RootFileCreatorFilterer(pair<string, string> data_holder,

30 string output_path,

33 bool debug)

35 // Enabling implicit Multi-threading
36 ROOT::EnableImplicitMT();
37
38 // Defining the quantities that will be used
39 ROOT::RDataFrame *dataset = NULL;
40
41 // Defining the name of the file and the name of the tree
42 char *filename = data_holder.first.data();
43 char *treename = data_holder.second.data();
44
45 // Purging the input filename from the input path and its extension to collect a pure name
46 string filename_clean = data_holder.first;
47 size_t inputpath_name_position = filename_clean.rfind("/");
48 if (inputpath_name_position != string::npos)
49 filename_clean.erase(0, inputpath_name_position+1);
50 size_t extension_name_position = filename_clean.find(".root");
51 if (extension_name_position != string::npos)
52 filename_clean.erase(extension_name_position, string(".root").length());
53 if (outputprint == false)
54 cout << filename_clean.data() << endl;
55
56 // Checking if the file and the tree do exist
57 bool filezombie = TFile(filename).IsZombie();
58 bool treezombie = TTree(filename, treename).IsZombie();
59
60 // If the file and the tree do exist, extract a RDataFrame
61 if (filezombie == false && treezombie == false)
62 dataset = new ROOT::RDataFrame(treename, filename);
63
64 // Filtering the new dataset using the cut given as a input to the function
65 ROOT::RDF::RNode dataset_filtered = dataset->Filter(cut);
66
67 // Looping over the columns that will be stored
68 vector<string> columnnames_todataset;
69 auto columnnames_fromdataset = dataset_filtered.GetColumnNames();
70 for (auto &&colname : columnnames_fromdataset)
71 columnnames_todataset.push_back(colname);
72
73 // Modifying write options of the root data frame to overwrite other trees
74 ROOT::RDF::RSnapshotOptions snapopt;
75 snapopt.fMode = "UPDATE";
76 snapopt.fOverwriteIfExists = "TRUE";
77
78 // Write the new dataset to a new file. The tree name is the original one
79 // Check if an output file name is specified, otherwise "_CREATED" is appended to the original one
80 if (output_filename.compare("")==0)
81 dataset_filtered.Snapshot(treename, (output_path+filename_clean+"_CREATED.root").data(), dataset_filtered.GetColumnNames(), snapopt);
82 else
83 dataset_filtered.Snapshot(treename, (output_path+output_filename+".root").data(), columnnames_todataset, snapopt);
84
85 // Disabling implicit Multi-threading
86 ROOT::DisableImplicitMT();
87
88 return 0;
89}
